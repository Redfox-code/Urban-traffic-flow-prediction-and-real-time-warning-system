### 3.2 TrafficMap — 地图主容器

```vue
<!-- frontend/src/components/map/TrafficMap.vue -->
<template>
  <div ref="mapContainer" class="traffic-map" :style="{ height: mapHeight }">
    <!-- 高德地图实例挂载点 -->
  </div>
  <!-- 子覆盖层 -->
  <SectionHeatmap v-if="mapReady" :map="mapInstance" :traffic-data="trafficData" />
  <VehicleTrajectory v-if="showTrajectory" :map="mapInstance" :trajectory-data="trajectoryData" />
  <MapMarker v-for="marker in markers" :key="marker.id" :map="mapInstance" :marker="marker" />
</template>

<script setup>
import { ref, onMounted, onUnmounted, watch } from 'vue'
import { useTrafficStore } from '@/store/traffic'

const props = defineProps({
  mapHeight: { type: String, default: 'calc(100vh - 64px)' },
  sections: { type: Array, default: () => [] },
})

const emit = defineEmits(['section-click'])

const mapContainer = ref(null)
const mapInstance = ref(null)
const mapReady = ref(false)
const trafficStore = useTrafficStore()

onMounted(async () => {
  // 动态加载高德JS API
  await loadAMapScript()
  mapInstance.value = new AMap.Map(mapContainer.value, {
    zoom: 13,
    center: [116.397, 39.908],
    mapStyle: 'amap://styles/darkblue',
    resizeEnable: true,
  })
  mapReady.value = true

  // 点击路段 → 联动ECharts图表
  mapInstance.value.on('click', (e) => {
    const section = findNearbySection(e.lnglat)
    if (section) emit('section-click', section)
  })
})

onUnmounted(() => {
  mapInstance.value?.destroy()
  mapInstance.value = null
})
</script>

<style scoped>
.traffic-map {
  width: 100%;
  border-radius: 8px;
  overflow: hidden;
}
</style>
```

## 4. WebSocket 客户端设计

### 4.1 连接架构

```javascript
// frontend/src/socketio/client.js
import { io } from 'socket.io-client'
import { useWarningStore } from '@/store/warning'
import { useTrafficStore } from '@/store/traffic'

const SOCKET_URL = import.meta.env.VITE_WS_URL || 'http://localhost:5000'

class SocketClient {
  constructor() {
    this.socket = null
    this.reconnectAttempts = 0
    this.maxReconnectAttempts = 10
  }

  connect(token) {
    this.socket = io(SOCKET_URL, {
      auth: { token },
      reconnection: true,
      reconnectionDelay: 1000,       // 初始重连延迟1s
      reconnectionDelayMax: 30000,   // 最大重连延迟30s
      reconnectionAttempts: this.maxReconnectAttempts,
      transports: ['websocket'],     // 只用WebSocket，不用HTTP长轮询
    })

    this.socket.on('connect', this.onConnect)
    this.socket.on('disconnect', this.onDisconnect)
    this.socket.on('connect_error', this.onError)

    // 业务事件
    this.socket.on('warning:new', this.onWarningNew)
    this.socket.on('warning:update', this.onWarningUpdate)
    this.socket.on('warning:resolve', this.onWarningResolve)
    this.socket.on('traffic:update', this.onTrafficUpdate)
  }

  // === 连接生命周期 ===
  onConnect() {
    console.log('[WS] 已连接')
    this.reconnectAttempts = 0
  }
  onDisconnect(reason) {
    console.warn('[WS] 断开:', reason)
  }
  onError(err) {
    console.error('[WS] 连接错误:', err.message)
  }

  // === 业务事件处理 ===
  onWarningNew(data) {
    const warningStore = useWarningStore()
    warningStore.addWarning(data)
    // 触发弹窗（由AlertPopup组件监听store变化）
  }
  onTrafficUpdate(data) {
    const trafficStore = useTrafficStore()
    trafficStore.updateRealtime(data)
  }

  // === 客户端 → 服务端 ===
  subscribeSection(sectionId) {
    this.socket?.emit('subscribe:section', { section_id: sectionId })
  }
  unsubscribeSection(sectionId) {
    this.socket?.emit('unsubscribe:section', { section_id: sectionId })
  }

  disconnect() {
    this.socket?.disconnect()
    this.socket = null
  }
}

// 单例导出
export const socketClient = new SocketClient()
```

## 5. 地图与ECharts联动

```
用户点击地图上的路段标记
  │
  ▼
TrafficMap → emit('section-click', { section_id, section_name })
  │
  ▼
父页面 (TrafficMonitor.vue) 接收事件
  │
  ├── 1. 更新 Pinia store (selectedSectionId)
  ├── 2. ECharts组件 watch(selectedSectionId) → 重新请求该路段的预测数据
  ├── 3. WebSocket订阅该路段(socketClient.subscribeSection(id))
  └── 4. 右侧信息面板显示该路段详情
```

```javascript
// 在 TrafficMonitor.vue 页面中
const handleSectionClick = (section) => {
  trafficStore.selectSection(section.section_id)
  socketClient.subscribeSection(section.section_id)
}

// ECharts 组件 watch Store 变化
watch(() => trafficStore.selectedSectionId, async (newId) => {
  if (!newId) return
  const data = await predictionApi.getForecast(newId)
  updateChart(data)
})
```
