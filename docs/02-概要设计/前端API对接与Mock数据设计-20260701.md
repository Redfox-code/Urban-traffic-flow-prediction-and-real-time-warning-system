## 1. API模块封装（7个模块）

基于 D4-T01 的30+端点，封装为7个前端API模块。

### 1.1 auth.js

```javascript
// frontend/src/api/auth.js
import request from './request'

export const authApi = {
  login: (data) => request.post('/auth/login', data),
  register: (data) => request.post('/auth/register', data),
  refresh: () => request.post('/auth/refresh'),
}
```

### 1.2 sections.js

```javascript
export const sectionsApi = {
  getList: (params) => request.get('/sections', { params }),
  getById: (id) => request.get(`/sections/${id}`),
  create: (data) => request.post('/sections', data),
  update: (id, data) => request.put(`/sections/${id}`, data),
  delete: (id) => request.delete(`/sections/${id}`),
}
```

### 1.3 traffic.js

```javascript
export const trafficApi = {
  getCurrent: (sectionId) => request.get('/traffic/current', { params: { section_id: sectionId } }),
  getHistory: (params) => request.get('/traffic/history', { params }),
}
```

### 1.4 prediction.js

```javascript
export const predictionApi = {
  getForecast: (sectionId, horizon = 15, model = 'RF') =>
    request.get('/predict/forecast', { params: { section_id: sectionId, horizon, model } }),
  getAccuracy: (sectionId, days = 7) =>
    request.get('/predict/accuracy', { params: { section_id: sectionId, days } }),
}
```

### 1.5 warning.js

```javascript
export const warningApi = {
  getList: (params) => request.get('/warning/list', { params }),
  resolve: (id) => request.put(`/warning/${id}/resolve`),
  getRules: () => request.get('/warning/rules'),
  updateRules: (data) => request.put('/warning/rules', data),
}
```

### 1.6 routePlan.js

```javascript
export const routeApi = {
  plan: (data) => request.post('/route/plan', data),
  getById: (id) => request.get(`/route/${id}`),
}
```

### 1.7 stats.js

```javascript
export const statsApi = {
  getDashboard: () => request.get('/stats/dashboard'),
  getDailyReport: (date) => request.get('/stats/daily_report', { params: { date } }),
}
```

### 2.3 Mock数据设计原则

| 原则 | 说明 |
|------|------|
| **覆盖全状态** | 每个端点Mock包含正常数据 + 空数据 + 错误场景 |
| **数据一致性** | section_id=1 在traffic/prediction/warning中代表同一路段 |
| **时间动态** | timestamp用`new Date()`而非固定值，Mock数据看起来像真实数据 |
| **分页真实** | total > page_size时返回不同数据，模拟真实翻页 |

## 3. 错误处理映射

基于 D4-T01 §1.4 错误码，前端统一映射为用户提示：

```javascript
// frontend/src/api/request.js — 响应拦截器增强

const ERROR_MAP = {
  400: '请求参数有误，请检查输入',
  401: '登录已过期，请重新登录',       // + 自动跳转登录页
  403: '无权限访问该功能',
  404: '请求的资源不存在',
  409: '数据冲突，请刷新后重试',
  422: '操作无法完成，请检查数据',
  500: '服务器繁忙，请稍后重试',
  'NETWORK_ERROR': '网络连接失败，请检查网络',
  'TIMEOUT': '请求超时，请稍后重试',
}

request.interceptors.response.use(
  (response) => {
    const { code, data } = response.data
    if (code === 200 || code === 201) return data
    return Promise.reject(response.data)
  },
  (error) => {
    if (error.response) {
      const status = error.response.status
      if (status === 401) { /* 跳转登录 */ }
      else if (status !== 422) { ElMessage.error(ERROR_MAP[status] || '请求失败') }
    } else if (error.code === 'ECONNABORTED') {
      ElMessage.error(ERROR_MAP.TIMEOUT)
    } else {
      ElMessage.error(ERROR_MAP.NETWORK_ERROR)
    }
    return Promise.reject(error)
  }
)
```
