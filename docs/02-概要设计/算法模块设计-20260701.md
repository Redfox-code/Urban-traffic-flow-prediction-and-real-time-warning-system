### 3.2 步骤详解

#### Step 1：SUMO 仿真运行与数据采集

```python
# algorithm/sumo/run_simulation.py
import traci
import pandas as pd

def run_simulation(sumo_cfg: str, duration: int = 3600) -> pd.DataFrame:
    """
    启动SUMO仿真，采集E2检测器数据
    输入：sumo配置文件路径，仿真时长（秒）
    输出：DataFrame [detector_id, timestamp, vehicle_count, avg_speed, occupancy]
    """
    records = []
    traci.start(["sumo", "-c", sumo_cfg])
    
    for step in range(duration):
        traci.simulationStep()
        if step % 900 == 0:  # 每15分钟采集一次
            for det_id in traci.lanearea.getIDList():
                records.append({
                    "detector_id": det_id,
                    "timestamp": step,
                    "vehicle_count": traci.lanearea.getJamLengthVehicle(det_id),
                    "avg_speed": traci.lanearea.getLastStepMeanSpeed(det_id),
                    "occupancy": traci.lanearea.getLastStepOccupancy(det_id),
                })
    
    traci.close()
    return pd.DataFrame(records)
```

#### Step 2：检测器数据解析

E2输出也支持从XML文件离线解析（批量仿真更高效）：

```python
# algorithm/preprocessor.py
def parse_e2_output(xml_path: str) -> pd.DataFrame:
    """解析SUMO e2_output.xml → 结构化DataFrame"""
    import xml.etree.ElementTree as ET
    tree = ET.parse(xml_path)
    # 提取 <interval> 元素，字段：begin, end, id, 
    #   sampledSeconds, traveltime, overlapTraveltime, ...
    records = []
    for interval in tree.findall(".//interval"):
        records.append({
            "detector_id": interval.get("id"),
            "begin": float(interval.get("begin")),
            "end": float(interval.get("end")),
            "vehicle_count": int(interval.get("sampledSeconds", 0)),
            "avg_speed": float(interval.get("speed", 0)),
            "occupancy": float(interval.get("occupancy", 0)),
        })
    return pd.DataFrame(records)
```

#### Step 3：缺失值与异常值处理

```python
def clean_data(df: pd.DataFrame) -> pd.DataFrame:
    """
    数据清洗Pipeline
    - 缺失值：同检测器前值填充 → 中位数填充
    - 异常值：IQR方法（Q1 - 1.5×IQR, Q3 + 1.5×IQR）
    - 负值修正：avg_speed < 0 → 0
    """
    df = df.copy()
    
    # 按检测器分组处理缺失值
    for col in ["vehicle_count", "avg_speed", "occupancy"]:
        df[col] = df.groupby("detector_id")[col].transform(
            lambda x: x.fillna(method="ffill").fillna(x.median())
        )
    
    # IQR异常值处理
    for col in ["vehicle_count", "avg_speed"]:
        Q1 = df[col].quantile(0.25)
        Q3 = df[col].quantile(0.75)
        IQR = Q3 - Q1
        df[col] = df[col].clip(Q1 - 1.5 * IQR, Q3 + 1.5 * IQR)
    
    # 负值修正
    df["avg_speed"] = df["avg_speed"].clip(lower=0)
    df["occupancy"] = df["occupancy"].clip(0, 100)
    
    return df
```

#### Step 4：特征工程

```python
def build_features(df: pd.DataFrame) -> pd.DataFrame:
    """
    构造预测特征：
    - 时间特征：hour, day_of_week, is_weekend, is_peak
    - 滞后特征：t-1, t-2, t-3 时刻的流量 (lag features)
    - 滚动统计：过去1小时均值、标准差 (rolling window)
    """
    df = df.copy()
    df["timestamp"] = pd.to_datetime(df["timestamp"])
    
    # 时间特征
    df["hour"] = df["timestamp"].dt.hour
    df["day_of_week"] = df["timestamp"].dt.dayofweek  # 0=Mon
    df["is_weekend"] = df["day_of_week"].isin([5, 6]).astype(int)
    df["is_peak"] = df["hour"].isin([7, 8, 17, 18]).astype(int)
    
    # 按检测器排序后构造滞后特征
    df = df.sort_values(["detector_id", "timestamp"])
    for lag in [1, 2, 3]:
        df[f"flow_lag_{lag}"] = df.groupby("detector_id")["vehicle_count"].shift(lag)
    
    # 滚动窗口统计（4步=1小时，假设15min采样）
    df["flow_rolling_mean_4"] = (
        df.groupby("detector_id")["vehicle_count"]
        .transform(lambda x: x.rolling(4, min_periods=1).mean())
    )
    df["flow_rolling_std_4"] = (
        df.groupby("detector_id")["vehicle_count"]
        .transform(lambda x: x.rolling(4, min_periods=1).std().fillna(0))
    )
    
    return df
```

```python
# algorithm/pipeline.py
def run_pipeline(sumo_cfg: str) -> tuple[pd.DataFrame, StandardScaler]:
    """
    一键执行完整数据管道
    返回：(特征矩阵X, 目标值y, scaler对象)
    """
    raw = run_simulation(sumo_cfg)        # Step 1
    df = parse_e2_output(raw) if xml else raw  # Step 2（离线则解析XML）
    df = clean_data(df)                   # Step 3
    df = build_features(df)               # Step 4
    
    target = "vehicle_count"
    feature_cols = [c for c in df.columns 
                    if c not in [target, "detector_id", "timestamp"]]
    
    X = df[feature_cols].dropna()
    y = df.loc[X.index, target]
    
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(X)
    X_scaled = pd.DataFrame(X_scaled, columns=feature_cols, index=X.index)
    
    return X_scaled, y, scaler
```

### 4.1 模型架构（策略模式）

```python
# algorithm/prediction/base_model.py
from abc import ABC, abstractmethod
import numpy as np

class BaseTrafficPredictor(ABC):
    """预测模型抽象基类 — 策略模式"""
    
    @abstractmethod
    def train(self, X: np.ndarray, y: np.ndarray) -> None:
        """训练模型"""
        pass
    
    @abstractmethod
    def predict(self, X: np.ndarray) -> np.ndarray:
        """预测"""
        pass
    
    @abstractmethod
    def get_params(self) -> dict:
        """获取模型超参数"""
        pass
```

### 4.2 KNN 回归模型

```python
# algorithm/prediction/knn_predictor.py
from sklearn.neighbors import KNeighborsRegressor
from sklearn.model_selection import GridSearchCV
import joblib

class KNNPredictor(BaseTrafficPredictor):
    """
    KNN回归预测器
    
    超参数：
    - n_neighbors: [3, 5, 7, 10, 15]
    - weights: ['uniform', 'distance']
    - metric: ['euclidean', 'manhattan']
    
    选型理由（ADR-002）：
    - 简单直观，无需训练时间
    - 交通流量有周期性（相似时间→相似流量），KNN天然匹配
    - 作为baseline模型，用于衡量RF的增益
    """
    
    def __init__(self):
        self.model = None
        self.best_params = None
    
    def train(self, X, y):
        param_grid = {
            "n_neighbors": [3, 5, 7, 10, 15],
            "weights": ["uniform", "distance"],
            "metric": ["euclidean", "manhattan"],
        }
        grid = GridSearchCV(
            KNeighborsRegressor(),
            param_grid,
            cv=5,
            scoring="neg_mean_absolute_error",
            n_jobs=-1,
        )
        grid.fit(X, y)
        self.model = grid.best_estimator_
        self.best_params = grid.best_params_
    
    def predict(self, X):
        return self.model.predict(X)
    
    def get_params(self):
        return self.best_params
    
    def save(self, path: str):
        joblib.dump(self.model, path)
    
    @staticmethod
    def load(path: str):
        predictor = KNNPredictor()
        predictor.model = joblib.load(path)
        return predictor
```

### 4.3 随机森林模型

```python
# algorithm/prediction/rf_predictor.py
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import GridSearchCV
import joblib

class RFPredictor(BaseTrafficPredictor):
    """
    随机森林预测器
    
    超参数：
    - n_estimators: [100, 200, 300]
    - max_depth: [10, 15, 20, None]
    - min_samples_split: [2, 5, 10]
    
    选型理由（ADR-002）：
    - 处理非线性特征交互能力强
    - 特征重要性可解释 → 分析哪些因素对流量影响最大
    - 不易过拟合（Bagging集成）
    """
    
    def __init__(self):
        self.model = None
        self.best_params = None
        self.feature_importance = None
    
    def train(self, X, y):
        param_grid = {
            "n_estimators": [100, 200],
            "max_depth": [10, 15, None],
            "min_samples_split": [2, 5],
            "min_samples_leaf": [1, 2],
        }
        grid = GridSearchCV(
            RandomForestRegressor(random_state=42, n_jobs=-1),
            param_grid,
            cv=5,
            scoring="neg_mean_absolute_error",
            n_jobs=-1,
        )
        grid.fit(X, y)
        self.model = grid.best_estimator_
        self.best_params = grid.best_params_
        self.feature_importance = dict(zip(
            X.columns if hasattr(X, "columns") else [f"f{i}" for i in range(X.shape[1])],
            self.model.feature_importances_
        ))
    
    def predict(self, X):
        return self.model.predict(X)
    
    def get_params(self):
        return self.best_params
    
    def get_feature_importance(self) -> dict:
        """返回特征重要性排序"""
        return dict(sorted(
            self.feature_importance.items(),
            key=lambda x: x[1], reverse=True
        ))
    
    def save(self, path: str):
        joblib.dump(self.model, path)
    
    @staticmethod
    def load(path: str):
        predictor = RFPredictor()
        predictor.model = joblib.load(path)
        return predictor
```

## 5. 模型评估方案

### 5.1 评估指标

| 指标 | 公式 | 说明 |
|------|------|------|
| **MAE** | `mean(|y_true - y_pred|)` | 平均绝对误差，单位：veh/15min |
| **RMSE** | `sqrt(mean((y_true - y_pred)²))` | 均方根误差，对大误差惩罚更重 |
| **MAPE** | `mean(|(y_true - y_pred)/y_true|) × 100` | 平均绝对百分比误差 |
| **R²** | `1 - SS_res/SS_tot` | 决定系数，越接近1越好 |

### 5.2 评估方法

```python
# algorithm/prediction/evaluator.py
from sklearn.model_selection import TimeSeriesSplit
from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score
import numpy as np

def evaluate_models(X, y, models: dict) -> pd.DataFrame:
    """
    时间序列交叉验证评估
    输入：特征X, 目标y, 模型字典{"KNN": knn_model, "RF": rf_model}
    输出：评估报告DataFrame
    """
    tscv = TimeSeriesSplit(n_splits=5)
    results = []
    
    for name, model in models.items():
        for fold, (train_idx, test_idx) in enumerate(tscv.split(X)):
            X_train, X_test = X.iloc[train_idx], X.iloc[test_idx]
            y_train, y_test = y.iloc[train_idx], y.iloc[test_idx]
            
            model.train(X_train.values, y_train.values)
            y_pred = model.predict(X_test.values)
            
            results.append({
                "model": name,
                "fold": fold + 1,
                "MAE": mean_absolute_error(y_test, y_pred),
                "RMSE": np.sqrt(mean_squared_error(y_test, y_pred)),
                "MAPE": np.mean(np.abs((y_test - y_pred) / (y_test + 1e-6))) * 100,
                "R2": r2_score(y_test, y_pred),
            })
    
    return pd.DataFrame(results)
```

| 指标 | KNN（期望） | RF（期望） | 合格阈值 |
|------|------------|-----------|---------|
| MAE | < 15 veh/15min | < 10 veh/15min | < 20 |
