### 2.2 导航守卫

```javascript
// 全局前置守卫
router.beforeEach((to, from, next) => {
  document.title = to.meta.title ? `${to.meta.title} | 智能交通系统` : '智能交通系统'
  
  const userStore = useUserStore()
  const isAuthenticated = !!userStore.token
  
  if (to.meta.requiresAuth !== false && !isAuthenticated) {
    // 未登录 → 跳转登录，记住目标路径
    next({ name: 'Login', query: { redirect: to.fullPath } })
  } 
  else if (to.meta.role === 'admin' && userStore.role !== 'admin') {
    // 非管理员访问admin路由 → 403
    next({ name: 'Dashboard' })
  } 
  else if ((to.name === 'Login' || to.name === 'Register') && isAuthenticated) {
    // 已登录访问登录页 → 重定向到首页
    next({ name: 'Dashboard' })
  } 
  else {
    next()
  }
})
```

### 3.2 useUserStore

```javascript
// frontend/src/store/user.js
import { defineStore } from 'pinia'
import { authApi } from '@/api/auth'

export const useUserStore = defineStore('user', {
  state: () => ({
    token: localStorage.getItem('token') || '',
    userInfo: null,
    role: '',
  }),
  
  getters: {
    isLoggedIn: (state) => !!state.token,
    isAdmin: (state) => state.role === 'admin',
  },
  
  actions: {
    async login(username, password) {
      const res = await authApi.login({ username, password })
      this.token = res.data.token
      this.userInfo = res.data.user
      this.role = res.data.user.role
      localStorage.setItem('token', this.token)
    },
    
    logout() {
      this.token = ''
      this.userInfo = null
      this.role = ''
      localStorage.removeItem('token')
    },
    
    async refreshToken() {
      try {
        const res = await authApi.refresh()
        this.token = res.data.token
        localStorage.setItem('token', this.token)
      } catch {
        this.logout()
      }
    },
  },
})
```

### 3.3 useTrafficStore

```javascript
// frontend/src/store/traffic.js
export const useTrafficStore = defineStore('traffic', {
  state: () => ({
    sections: [],           // 全部路段列表
    selectedSectionId: null,// 当前选中路段
    realtimeData: {},       // { [sectionId]: { vehicle_count, avg_speed, occupancy, timestamp } }
    isLoading: false,
  }),
  
  actions: {
    async fetchSections() {
      this.isLoading = true
      const res = await trafficApi.getSections()
      this.sections = res.data.items
      this.isLoading = false
    },
    
    selectSection(sectionId) {
      this.selectedSectionId = sectionId
    },
    
    updateRealtime(data) {
      // 由 WebSocket 或 API 调用
      this.realtimeData[data.section_id] = {
        vehicleCount: data.vehicle_count,
        avgSpeed: data.avg_speed,
        occupancy: data.occupancy,
        timestamp: data.timestamp,
      }
    },
  },
})
```

### 3.4 useWarningStore

```javascript
// frontend/src/store/warning.js
export const useWarningStore = defineStore('warning', {
  state: () => ({
    warnings: [],
    unreadCount: 0,
    filter: { level: '', resolved: false, sectionId: null },
    isLoading: false,
  }),
  
  getters: {
    filteredWarnings: (state) => {
      let list = state.warnings
      if (state.filter.level) list = list.filter(w => w.level === state.filter.level)
      if (state.filter.resolved !== null) list = list.filter(w => w.is_resolved === state.filter.resolved)
      if (state.filter.sectionId) list = list.filter(w => w.section_id === state.filter.sectionId)
      return list
    },
    activeWarnings: (state) => state.warnings.filter(w => !w.is_resolved),
  },
  
  actions: {
    async fetchWarnings(params = {}) {
      const res = await warningApi.getList(params)
      this.warnings = res.data.items
      this.unreadCount = this.activeWarnings.length
    },
    
    addWarning(warning) {
      // 由 FE-Map 的 WebSocket 客户端调用
      this.warnings.unshift(warning)
      if (!warning.is_resolved) this.unreadCount++
    },
    
    async resolveWarning(id) {
      await warningApi.resolve(id)
      const w = this.warnings.find(w => w.id === id)
      if (w) { w.is_resolved = true; this.unreadCount-- }
    },
  },
})
```

## 4. Axios API 封装

### 4.1 请求实例

```javascript
// frontend/src/api/request.js
import axios from 'axios'
import { ElMessage } from 'element-plus'
import { useUserStore } from '@/store/user'

const request = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || 'http://localhost:5000/api/v1',
  timeout: 15000,
  headers: { 'Content-Type': 'application/json' },
})

// 请求拦截器 — JWT注入
request.interceptors.request.use(
  (config) => {
    const userStore = useUserStore()
    if (userStore.token) {
      config.headers.Authorization = `Bearer ${userStore.token}`
    }
    return config
  },
  (error) => Promise.reject(error)
)

// 响应拦截器 — 统一错误处理
request.interceptors.response.use(
  (response) => {
    const { code, message, data } = response.data
    if (code === 200 || code === 201) return data  // 解包统一返回格式
    ElMessage.error(message || '请求失败')
    return Promise.reject(new Error(message))
  },
  (error) => {
    if (error.response?.status === 401) {
      const userStore = useUserStore()
      userStore.logout()
      ElMessage.error('登录已过期，请重新登录')
      router.push({ name: 'Login', query: { redirect: router.currentRoute.value.fullPath } })
    } else if (error.response?.status === 403) {
      ElMessage.error('无权限访问')
    } else if (error.response?.status >= 500) {
      ElMessage.error('服务器错误，请稍后重试')
    } else if (error.code === 'ECONNABORTED') {
      ElMessage.error('请求超时')
    }
    return Promise.reject(error)
  }
)

export default request
```

### 5.2 可复用图表组件

```vue
<!-- frontend/src/components/charts/BaseChart.vue -->
<template>
  <div ref="chartRef" :style="{ width: '100%', height: height }"></div>
</template>

<script setup>
import { ref, inject, onMounted, onUnmounted, watch } from 'vue'

const props = defineProps({
  option: { type: Object, required: true },
  height: { type: String, default: '400px' },
})

const echarts = inject('echarts')
const chartRef = ref(null)
let chartInstance = null

onMounted(() => {
  chartInstance = echarts.init(chartRef.value, 'dark')  // 暗色主题配大屏
  chartInstance.setOption(props.option)
  window.addEventListener('resize', () => chartInstance?.resize())
})

watch(() => props.option, (newOpt) => {
  chartInstance?.setOption(newOpt, true)  // true = notMerge, 完整替换
}, { deep: true })

onUnmounted(() => {
  chartInstance?.dispose()
  chartInstance = null
})
</script>
```
